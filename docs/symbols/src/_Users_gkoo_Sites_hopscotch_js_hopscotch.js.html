 55     /**
 56      * addClass
 57      * ========
 58      * Adds a class to a DOM element.
 59      * Note: does not support adding multiple classes at once yet
 60      */
 61     addClass: function(domEl, classToAdd) {
 62       var domClasses,
 63           i, len;
 64 
 65       if (domEl.className.length === 0) {
 66         domEl.className = classToAdd;
 67       }
 68       else {
 69         domClasses = domEl.className.split(' ');
 70         for (i = 0, len = domClasses.length; i < len; ++i) {
 71           if (domClasses[i] === classToAdd) {
 72             return;
 73           }
 74         }
 75         domClasses.splice(0, 0, classToAdd); // add new class to list
 76         domEl.className = domClasses.join(' ');
 77       }
 78     },
 79 
 80     /**
 81      * removeClass
 82      * ===========
 83      * Remove a class from a DOM element.
 84      * Note: this one DOES support removing multiple classes.
 85      */
 86     removeClass: function(domEl, classToRemove) {
 87       var domClasses,
 88           classesToRemove,
 89           currClass,
 90           i,
 91           j,
 92           toRemoveLen,
 93           domClassLen;
 94 
 95       classesToRemove = classToRemove.split(' ');
 96       domClasses = domEl.className.split(' ');
 97       for (i = 0, toRemoveLen = classesToRemove.length; i < toRemoveLen; ++i) {
 98         currClass = classesToRemove[i];
 99         for (j = 0, domClassLen = domClasses.length; j < domClassLen; ++j) {
100           if (domClasses[j] === currClass) {
101             break;
102           }
103         }
104         if (j < domClassLen) {
105           domClasses.splice(j, 1); // remove class from list
106         }
107       }
108       domEl.className = domClasses.join(' ');
109     },

191     // Tour session persistence for multi-page tours. Uses HTML5 sessionStorage if available, then
192     // falls back to using cookies.
193     //
194     // The following cookie-related logic is borrowed from:
195     // http://www.quirksmode.org/js/cookies.html

197     setState: function(name,value,days) {
198       var expires = '',
199           date;
200 
201       if (hasSessionStorage) {
202         sessionStorage.setItem(name, value);
203       }
204       else {
205         if (days) {
206           date = new Date();
207           date.setTime(date.getTime()+(days*24*60*60*1000));
208           expires = "; expires="+date.toGMTString();
209         }
210         document.cookie = name+"="+value+expires+"; path=/";
211       }
212     },
213 
214     getState: function(name) {
215       var nameEQ = name + "=",
216           ca = document.cookie.split(';'),
217           i,
218           c;
219 
220       if (hasSessionStorage) {
221         return sessionStorage.getItem(name);
222       }
223       else {
224         for(i=0;i < ca.length;i++) {
225           c = ca[i];
226           while (c.charAt(0)===' ') c = c.substring(1,c.length);
227           if (c.indexOf(nameEQ) === 0) return c.substring(nameEQ.length,c.length);
228         }
229         return null;
230       }
231     },

233     clearState: function(name) {
234       if (hasSessionStorage) {
235         sessionStorage.removeItem(name);
236       }
237       else {
238         this.setState(name,"",-1);
239       }

309     /**
310      * setPosition
311      *
312      * Sets the position of the bubble using the bounding rectangle of the
313      * target element and the orientation and offset information specified by
314      * the step JSON.
315      *
316      * @private
317      */
318     setPosition = function(step) {
319       var bubbleWidth,
320           bubbleHeight,
321           bubblePadding,
322           boundingRect,
323           top,
324           left,
325           bubbleBorder = 6,
326           targetEl     = utils.getStepTarget(step),
327           el           = this.element,
328           arrowEl      = this.arrowEl,
329           arrowOffset  = utils.getPixelValue(step.arrowOffset);
330 
331       bubbleWidth   = utils.getPixelValue(step.width) || opt.bubbleWidth;
332       bubblePadding = utils.valOrDefault(step.padding, opt.bubblePadding);
333 
334       utils.removeClass(el, 'fade-in-down fade-in-up fade-in-left fade-in-right');
335 
336       // SET POSITION
337       boundingRect = targetEl.getBoundingClientRect();
338       if (step.orientation === 'top') {
339         bubbleHeight = el.offsetHeight;
340         top = (boundingRect.top - bubbleHeight) - opt.arrowWidth;
341         left = boundingRect.left;
342       }
343       else if (step.orientation === 'bottom') {
344         top = boundingRect.bottom + opt.arrowWidth;
345         left = boundingRect.left;
346       }
347       else if (step.orientation === 'left') {
348         top = boundingRect.top;
349         left = boundingRect.left - bubbleWidth - 2*bubblePadding - 2*bubbleBorder - opt.arrowWidth;
350       }
351       else if (step.orientation === 'right') {
352         top = boundingRect.top;
353         left = boundingRect.right + opt.arrowWidth;
354       }
355 
356       // SET (OR RESET) ARROW OFFSETS
357       if (!arrowOffset) {
358         arrowEl.style.top = '';
359         arrowEl.style.left = '';
360       }
361       else if (step.orientation === 'top' || step.orientation === 'bottom') {
362         arrowEl.style.top = '';
363         arrowEl.style.left = arrowOffset + 'px';
364       }
365       else if (step.orientation === 'left' || step.orientation === 'right') {
366         arrowEl.style.left = '';
367         arrowEl.style.top = arrowOffset + 'px';
368       }
369 
370       // SET OFFSETS
371       left += utils.getPixelValue(step.xOffset);
372       top += utils.getPixelValue(step.yOffset);
373 
374       // ADJUST TOP FOR SCROLL POSITION
375       if (!step.fixedElement) {
376         top += utils.getScrollTop();
377         left += utils.getScrollLeft();
378       }
379 
380       // ACCOUNT FOR FIXED POSITION ELEMENTS
381       el.style.position = (step.fixedElement ? 'fixed' : 'absolute');
382 
383       if (opt.animate && hasJquery && !hasCssTransitions) {
384         $(el).animate({
385           top: top + 'px',
386           left: left + 'px'
387         });
388       }
389       else {
390         el.style.top = top + 'px';
391         el.style.left = left + 'px';
392       }
393     },

726     /**
727      * initAnimate
728      * ===========
729      * This function exists due to how Chrome handles initial CSS transitions.
730      * Most other browsers will not animate a transition until the element
731      * exists on the page. Chrome treats DOM elements as starting from the
732      * (0, 0) position, and will animate from the upper left corner on creation
733      * of the DOM element. (e.g., if you create a new DOM element using
734      * Javascript and specify CSS top: 100px, left: 50px, then append the
735      * DOM element to the document.body, it will create it at 0, 0 and then
736      * animate it to 50, 100)
737      *
738      * Solution is to add the animate class (which defines our transition)
739      * only after the element is created.
740      */
741     this.initAnimate = function() {
742       var self = this;
743       setTimeout(function() {
744         utils.addClass(self.element, 'animate');
745       }, 50);
746     };

865     /**
866      * adjustWindowScroll
867      *
868      * Checks if the bubble or target element is partially or completely
869      * outside of the viewport. If it is, adjust the window scroll position
870      * to bring it back into the viewport.
871      *
872      * @private
873      * @param {Function} cb Callback to invoke after done scrolling.
874      */
875     adjustWindowScroll = function(cb) {
876       var bubble         = getBubble(),
877 
878           // Calculate the bubble element top and bottom position
879           bubbleEl       = bubble.element,
880           bubbleTop      = utils.getPixelValue(bubbleEl.style.top),
881           bubbleBottom   = bubbleTop + utils.getPixelValue(bubbleEl.offsetHeight),
882 
883           // Calculate the target element top and bottom position
884           targetEl       = utils.getStepTarget(getCurrStep()),
885           targetBounds   = targetEl.getBoundingClientRect(),
886           targetElTop    = targetBounds.top + utils.getScrollTop(),
887           targetElBottom = targetBounds.bottom + utils.getScrollTop(),
888 
889           // The higher of the two: bubble or target
890           targetTop      = (bubbleTop < targetElTop) ? bubbleTop : targetElTop,
891           // The lower of the two: bubble or target
892           targetBottom   = (bubbleBottom > targetElBottom) ? bubbleBottom : targetElBottom,
893 
894           // Calculate the current viewport top and bottom
895           windowTop      = utils.getScrollTop(),
896           windowBottom   = windowTop + utils.getWindowHeight(),
897 
898           // This is our final target scroll value.
899           scrollToVal    = targetTop - opt.scrollTopMargin,
900 
901           self           = this,
902           scrollEl,
903           yuiAnim,
904           yuiEase,
905           direction,
906           scrollIncr,
907           scrollInt;
908 
909       // Target and bubble are both visible in viewport
910       if (targetTop >= windowTop && (targetTop <= windowTop + opt.scrollTopMargin || targetBottom <= windowBottom)) {
911         if (cb) { cb(); } // HopscotchBubble.show
912         return;
913       }
914 
915       // Abrupt scroll to scroll target
916       else if (!opt.smoothScroll) {
917         window.scrollTo(0, scrollToVal);
918 
919         if (cb) { cb(); } // HopscotchBubble.show
920         return;
921       }
922 
923       // Smooth scroll to scroll target
924       else {
925         // Use YUI if it exists
926         if (typeof YAHOO             !== undefinedStr &&
927             typeof YAHOO.env         !== undefinedStr &&
928             typeof YAHOO.env.ua      !== undefinedStr &&
929             typeof YAHOO.util        !== undefinedStr &&
930             typeof YAHOO.util.Scroll !== undefinedStr) {
931           scrollEl = YAHOO.env.ua.webkit ? document.body : document.documentElement;
932           yuiEase = YAHOO.util.Easing ? YAHOO.util.Easing.easeOut : undefined;
933           yuiAnim = new YAHOO.util.Scroll(scrollEl, {
934             scroll: { to: [0, scrollToVal] }
935           }, opt.scrollDuration/1000, yuiEase);
936           yuiAnim.onComplete.subscribe(cb);
937           yuiAnim.animate();
938         }
939 
940         // Use jQuery if it exists
941         else if (hasJquery) {
942           $('body, html').animate({ scrollTop: scrollToVal }, opt.scrollDuration, cb);
943         }
944 
945         // Use my crummy setInterval scroll solution if we're using plain, vanilla Javascript.
946         else {
947           if (scrollToVal < 0) {
948             scrollToVal = 0;
949           }
950 
951           // 48 * 10 == 480ms scroll duration
952           // make it slightly less than CSS transition duration because of
953           // setInterval overhead.
954           // To increase or decrease duration, change the divisor of scrollIncr.
955           direction = (windowTop > targetTop) ? -1 : 1; // -1 means scrolling up, 1 means down
956           scrollIncr = Math.abs(windowTop - scrollToVal) / (opt.scrollDuration/10);
957           scrollInt = setInterval(function() {
958             var scrollTop = utils.getScrollTop(),
959                 scrollTarget = scrollTop + (direction * scrollIncr);
960 
961             if ((direction > 0 && scrollTarget >= scrollToVal) ||
962                 direction < 0 && scrollTarget <= scrollToVal) {
963               // Overshot our target. Just manually set to equal the target
964               // and clear the interval
965               scrollTarget = scrollToVal;
966               clearInterval(scrollInt);
967               if (cb) { cb(); } // HopscotchBubble.show
968               window.scrollTo(0, scrollTarget);
969               return;
970             }
971 
972             window.scrollTo(0, scrollTarget);
973 
974             if (utils.getScrollTop() === scrollTop) {
975               // Couldn't scroll any further. Clear interval.
976               clearInterval(scrollInt);
977 
978               if (cb) { cb(); } // HopscotchBubble.show
979             }
980           }, 10);
981         }
982       }
983     },

987      *
988      * Helper function to change step by going forwards or backwards 1.
989      * nextStep and prevStep are publicly accessible wrappers for this function.

993      * @param {Number} direction Either 1 for "next" or -1 for "prev"

998           bubble      = getBubble();

1011             currStepNum += direction;

1134      *

1136      *
1137      * @param {Object} tour The tour JSON object
1138      * @stepNum {Number} stepNum __Optional__ The step number to start from

1143     this.startTour = function(tour, stepNum, substepNum) {

1155         currSubstepNum = substepNum;

1161         waitingToStart = true;
1162         return this;
1163       }
1164 
1165       if (typeof currStepNum === undefinedStr) {

1192         // Multi-part step

1195       */

1485           cbArr = callbacks[evtType];
1486           for (i=0, len=cbArr.length; i < len; ++i) {
1487             if (cbArr[i].fromTour) {
1488               cbArr.splice(i--, 1);
1489               --len;
1490             }
1491           }
1492         }
1493         else {
1494           callbacks[evtType] = [];
1495         }
1496       }
1497       return this;
1498     };
